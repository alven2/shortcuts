-configurar usuario local
git config --global user.name "alven"
git config -- global user.email "ppokevaz@gmail.com"

-creando repositorio (path de projecto)
git init

-conectar con un repo en github
git remote add origin "url"

-subir rama a github
git push origin master
-verificar el estado de repo
git status

-seleccionar todos los archivos del repo para precommitear y ser detectado
git add .

-para commit (el -m es mensaje y es obli)
git commit -m "hice este cambio"

-para ver el historial de commits
git log

-para cargar commit viejos or, nuevos
git checkout "codigo del log"

-para verificar la rama que estas actual
git branch

-para cambiar a la rama base
git switch master
